# MCP Jira Core MVP — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Поднять single-codebase / два бинаря Go-проекта `mcp-jira` (сервер) и `mcp-jira-index` (CLI), реализующих три MCP tools поверх Jira — `list_issues`, `get_sprint_health`, `search_jira_knowledge` (RAG через Postgres+pgvector и Voyage/OpenAI embeddings), со stdio и Streamable HTTP транспортами.

**Architecture:** Строгая однонаправленная зависимость `cmd → register → handlers → (jira|rag/retriever) → (rag/store|rag/embed)`. `internal/handlers` не импортирует MCP SDK. `internal/register` — единственный импортёр `go-sdk/mcp` и содержит `adapt[In,Out]`, приводящий `Handler[In,Out] func(ctx, In) (Out, error)` к сигнатуре MCP SDK `func(ctx, *mcp.CallToolRequest, In) (*mcp.CallToolResult, Out, error)`. HTTP — только в `cmd/server` через Echo v4.

**Tech Stack:** Go 1.26+, `github.com/modelcontextprotocol/go-sdk`, `github.com/labstack/echo/v4`, `github.com/jackc/pgx/v5`, `github.com/pressly/goose/v3`, `github.com/stretchr/testify/require`, `github.com/testcontainers/testcontainers-go` (+ `modules/postgres`). Jira REST API v3 (`/rest/api/3/search/jql`) + Agile API v1.0 через `net/http`. Voyage AI (default) и OpenAI для embeddings, размерность фиксирована `vector(1024)`.

**Spec:** `docs/superpowers/specs/2026-04-06-mcp-jira-design.md`. Перед работой прочитай его целиком.

```go
package config

import (
	"fmt"
	"os"
)

type Mode int

const (
	ModeStdio Mode = iota
	ModeHTTP
	ModeIndex
)

type Config struct {
	JiraBaseURL  string
	JiraEmail    string
	JiraAPIToken string

	DatabaseURL string

	RAGEmbedder  string // "voyage" | "openai"
	VoyageAPIKey string
	OpenAIAPIKey string

	MCPAPIKey string
	MCPAddr   string
}

func Load(mode Mode) (Config, error) {
	cfg := Config{
		JiraBaseURL:  os.Getenv("JIRA_BASE_URL"),
		JiraEmail:    os.Getenv("JIRA_EMAIL"),
		JiraAPIToken: os.Getenv("JIRA_API_TOKEN"),
		DatabaseURL:  os.Getenv("DATABASE_URL"),
		RAGEmbedder:  os.Getenv("RAG_EMBEDDER"),
		VoyageAPIKey: os.Getenv("VOYAGE_API_KEY"),
		OpenAIAPIKey: os.Getenv("OPENAI_API_KEY"),
		MCPAPIKey:    os.Getenv("MCP_API_KEY"),
		MCPAddr:      os.Getenv("MCP_ADDR"),
	}
	if cfg.RAGEmbedder == "" {
		cfg.RAGEmbedder = "voyage"
	}
	if cfg.MCPAddr == "" {
		cfg.MCPAddr = ":8080"
	}

	if cfg.JiraBaseURL == "" {
		return Config{}, fmt.Errorf("config: JIRA_BASE_URL is required")
	}
	if cfg.JiraEmail == "" {
		return Config{}, fmt.Errorf("config: JIRA_EMAIL is required")
	}
	if cfg.JiraAPIToken == "" {
		return Config{}, fmt.Errorf("config: JIRA_API_TOKEN is required")
	}
	if cfg.DatabaseURL == "" {
		return Config{}, fmt.Errorf("config: DATABASE_URL is required")
	}

	switch cfg.RAGEmbedder {
	case "voyage":
		if cfg.VoyageAPIKey == "" {
			return Config{}, fmt.Errorf("config: VOYAGE_API_KEY is required for RAG_EMBEDDER=voyage")
		}
	case "openai":
		if cfg.OpenAIAPIKey == "" {
			return Config{}, fmt.Errorf("config: OPENAI_API_KEY is required for RAG_EMBEDDER=openai")
		}
	default:
		return Config{}, fmt.Errorf("config: RAG_EMBEDDER must be 'voyage' or 'openai', got %q", cfg.RAGEmbedder)
	}

	if mode == ModeHTTP && cfg.MCPAPIKey == "" {
		return Config{}, fmt.Errorf("config: MCP_API_KEY is required for http mode")
	}

	return cfg, nil
}
```

```go
package jira

import "strings"

// QuoteJQL экранирует строку для вставки в JQL в виде "value".
// Правила: \ → \\, " → \", затем оборачиваем в двойные кавычки.
// Порядок важен: сначала обрабатываем backslash, потом кавычки.
func QuoteJQL(s string) string {
	s = strings.ReplaceAll(s, `\`, `\\`)
	s = strings.ReplaceAll(s, `"`, `\"`)
	return `"` + s + `"`
}
```

```go
package jira

import (
	"context"
	"encoding/json"
	"fmt"
	"net/http"
	"net/url"
	"strconv"
	"strings"
	"time"
)

type HTTPClient struct {
	baseURL string
	email   string
	token   string
	http    *http.Client
}

func NewHTTPClient(baseURL, email, token string) *HTTPClient {
	return &HTTPClient{
		baseURL: strings.TrimRight(baseURL, "/"),
		email:   email,
		token:   token,
		http:    &http.Client{Timeout: 30 * time.Second},
	}
}

func (c *HTTPClient) doJSON(ctx context.Context, method, path string, q url.Values, out any) error {
	u := c.baseURL + path
	if len(q) > 0 {
		u += "?" + q.Encode()
	}
	req, err := http.NewRequestWithContext(ctx, method, u, nil)
	if err != nil {
		return err
	}
	req.SetBasicAuth(c.email, c.token)
	req.Header.Set("Accept", "application/json")
	resp, err := c.http.Do(req)
	if err != nil {
		return fmt.Errorf("jira: %s %s: %w", method, path, err)
	}
	defer resp.Body.Close()
	if resp.StatusCode >= 400 {
		return fmt.Errorf("jira: %s %s -> %d", method, path, resp.StatusCode)
	}
	if out == nil {
		return nil
	}
	return json.NewDecoder(resp.Body).Decode(out)
}

func (c *HTTPClient) ListIssues(ctx context.Context, p ListIssuesParams) ([]Issue, error) {
	parts := []string{"project = " + QuoteJQL(p.ProjectKey)}
	if p.Status != "" {
		parts = append(parts, "status = "+QuoteJQL(p.Status))
	}
	if p.Assignee != "" {
		parts = append(parts, "assignee = "+QuoteJQL(p.Assignee))
	}
	limit := p.Limit
	if limit <= 0 {
		limit = 25
	}
	if limit > 100 {
		limit = 100
	}

	q := url.Values{
		"jql":        {strings.Join(parts, " AND ")},
		"fields":     {"summary,status,assignee"},
		"maxResults": {strconv.Itoa(limit)},
	}

	var raw struct {
		Issues []struct {
			Key    string `json:"key"`
			Fields struct {
				Summary string `json:"summary"`
				Status  struct {
					Name string `json:"name"`
				} `json:"status"`
				Assignee *struct {
					DisplayName string `json:"displayName"`
				} `json:"assignee"`
			} `json:"fields"`
		} `json:"issues"`
	}
	if err := c.doJSON(ctx, http.MethodGet, "/rest/api/3/search/jql", q, &raw); err != nil {
		return nil, err
	}
	out := make([]Issue, 0, len(raw.Issues))
	for _, it := range raw.Issues {
		issue := Issue{Key: it.Key, Summary: it.Fields.Summary, Status: it.Fields.Status.Name}
		if it.Fields.Assignee != nil {
			issue.Assignee = it.Fields.Assignee.DisplayName
		}
		out = append(out, issue)
	}
	return out, nil
}
```

```go
func (c *HTTPClient) IterateIssueDocs(ctx context.Context, projectKey string) (<-chan IssueDoc, <-chan error) {
	docs := make(chan IssueDoc, 16)
	errs := make(chan error, 1)

	go func() {
		defer close(docs)
		defer close(errs)

		var nextToken string
		for {
			q := url.Values{
				"jql":        {"project = " + QuoteJQL(projectKey) + " ORDER BY updated DESC"},
				"fields":     {"summary,status,assignee,description,issuelinks,updated"},
				"expand":     {"changelog"},
				"maxResults": {"100"},
			}
			if nextToken != "" {
				q.Set("nextPageToken", nextToken)
			}

			var page struct {
				NextPageToken string `json:"nextPageToken"`
				Issues        []struct {
					Key    string `json:"key"`
					Fields struct {
						Summary  string `json:"summary"`
						Status   struct{ Name string } `json:"status"`
						Assignee *struct{ DisplayName string } `json:"assignee"`
						// Description может быть строкой или ADF-объектом; принимаем raw и достаём строку если она есть.
						Description json.RawMessage `json:"description"`
						Updated     string          `json:"updated"`
						IssueLinks  []struct {
							OutwardIssue *struct{ Key string } `json:"outwardIssue"`
							InwardIssue  *struct{ Key string } `json:"inwardIssue"`
						} `json:"issuelinks"`
					} `json:"fields"`
					Changelog struct {
						Histories []struct {
							Created string `json:"created"`
							Items   []struct {
								Field      string `json:"field"`
								FromString string `json:"fromString"`
								ToString   string `json:"toString"`
							} `json:"items"`
						} `json:"histories"`
					} `json:"changelog"`
				} `json:"issues"`
			}
			if err := c.doJSON(ctx, http.MethodGet, "/rest/api/3/search/jql", q, &page); err != nil {
				errs <- err
				return
			}

			for _, it := range page.Issues {
				doc := IssueDoc{
					ProjectKey:  projectKey,
					Key:         it.Key,
					Summary:     it.Fields.Summary,
					Status:      it.Fields.Status.Name,
					Description: decodeDescription(it.Fields.Description),
				}
				if it.Fields.Assignee != nil {
					doc.Assignee = it.Fields.Assignee.DisplayName
				}
				if t, err := time.Parse("2006-01-02T15:04:05.000-0700", it.Fields.Updated); err == nil {
					doc.UpdatedAt = t
				}
				for _, l := range it.Fields.IssueLinks {
					if l.OutwardIssue != nil {
						doc.LinkedIssues = append(doc.LinkedIssues, l.OutwardIssue.Key)
					}
					if l.InwardIssue != nil {
						doc.LinkedIssues = append(doc.LinkedIssues, l.InwardIssue.Key)
					}
				}
				for _, h := range it.Changelog.Histories {
					for _, item := range h.Items {
						if item.Field == "status" {
							doc.StatusHistory = append(doc.StatusHistory,
								fmt.Sprintf("%s: %s → %s", h.Created, item.FromString, item.ToString))
						}
					}
				}

				comments, err := c.fetchComments(ctx, it.Key)
				if err != nil {
					errs <- err
					return
				}
				doc.Comments = comments

				select {
				case docs <- doc:
				case <-ctx.Done():
					errs <- ctx.Err()
					return
				}
			}

			if page.NextPageToken == "" {
				return
			}
			nextToken = page.NextPageToken
		}
	}()

	return docs, errs
}

func (c *HTTPClient) fetchComments(ctx context.Context, issueKey string) ([]string, error) {
	var raw struct {
		Comments []struct {
			Body json.RawMessage `json:"body"`
		} `json:"comments"`
	}
	path := "/rest/api/3/issue/" + issueKey + "/comment"
	if err := c.doJSON(ctx, http.MethodGet, path, nil, &raw); err != nil {
		return nil, err
	}
	out := make([]string, 0, len(raw.Comments))
	for _, c := range raw.Comments {
		if s := decodeDescription(c.Body); s != "" {
			out = append(out, s)
		}
	}
	return out, nil
}

// decodeDescription принимает поле, которое в API v3 может быть строкой или ADF-объектом.
// Для MVP: строка → как есть; объект → JSON-сериализация в строку (для embedding достаточно).
func decodeDescription(raw json.RawMessage) string {
	if len(raw) == 0 {
		return ""
	}
	var s string
	if err := json.Unmarshal(raw, &s); err == nil {
		return s
	}
	return string(raw)
}
```

```go
package auth

import (
	"crypto/subtle"
	"net/http"
)

// Middleware возвращает stdlib-совместимое middleware, проверяющее X-API-Key
// через constant-time сравнение. Пустой ключ клиента всегда отвергается.
func Middleware(expected string) func(http.Handler) http.Handler {
	exp := []byte(expected)
	return func(next http.Handler) http.Handler {
		return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
			got := []byte(r.Header.Get("X-API-Key"))
			if len(got) == 0 || subtle.ConstantTimeCompare(got, exp) != 1 {
				http.Error(w, "unauthorized", http.StatusUnauthorized)
				return
			}
			next.ServeHTTP(w, r)
		})
	}
}
```

```go
package store

import (
	"context"
	"fmt"
	"strconv"
	"strings"

	"github.com/jackc/pgx/v5/pgxpool"
)

type PgvectorStore struct {
	pool *pgxpool.Pool
}

func NewPgvectorStore(ctx context.Context, dsn string) (*PgvectorStore, error) {
	pool, err := pgxpool.New(ctx, dsn)
	if err != nil {
		return nil, fmt.Errorf("pgvector: connect: %w", err)
	}
	if err := pool.Ping(ctx); err != nil {
		pool.Close()
		return nil, fmt.Errorf("pgvector: ping: %w", err)
	}
	return &PgvectorStore{pool: pool}, nil
}

func (s *PgvectorStore) Close() {
	s.pool.Close()
}

// vectorLiteral сериализует []float32 в строковый литерал pgvector "[0.1,0.2,...]".
func vectorLiteral(v []float32) string {
	var b strings.Builder
	b.WriteByte('[')
	for i, x := range v {
		if i > 0 {
			b.WriteByte(',')
		}
		b.WriteString(strconv.FormatFloat(float64(x), 'f', -1, 32))
	}
	b.WriteByte(']')
	return b.String()
}

func (s *PgvectorStore) ReplaceProject(ctx context.Context, projectKey string, docs []Document) error {
	tx, err := s.pool.Begin(ctx)
	if err != nil {
		return fmt.Errorf("pgvector: begin: %w", err)
	}
	defer func() { _ = tx.Rollback(ctx) }()

	if _, err := tx.Exec(ctx, `DELETE FROM issues_index WHERE project_key = $1`, projectKey); err != nil {
		return fmt.Errorf("pgvector: delete: %w", err)
	}

	const insert = `
INSERT INTO issues_index (project_key, issue_key, summary, status, assignee, content, embedding)
VALUES ($1, $2, $3, $4, $5, $6, $7::vector)
`
	for _, d := range docs {
		if len(d.Embedding) == 0 {
			return fmt.Errorf("pgvector: empty embedding for %s", d.IssueKey)
		}
		if _, err := tx.Exec(ctx, insert,
			d.ProjectKey, d.IssueKey, d.Summary, d.Status, d.Assignee, d.Content, vectorLiteral(d.Embedding),
		); err != nil {
			return fmt.Errorf("pgvector: insert %s: %w", d.IssueKey, err)
		}
	}

	if err := tx.Commit(ctx); err != nil {
		return fmt.Errorf("pgvector: commit: %w", err)
	}
	return nil
}

func (s *PgvectorStore) Query(ctx context.Context, vec []float32, topK int, f Filter) ([]Hit, error) {
	if topK <= 0 {
		topK = 5
	}
	if topK > 50 {
		topK = 50
	}
	const q = `
SELECT issue_key, summary, status,
       1 - (embedding <=> $1::vector) AS score,
       substring(content, 1, 300) AS excerpt
FROM issues_index
WHERE project_key = $2
ORDER BY embedding <=> $1::vector
LIMIT $3
`
	rows, err := s.pool.Query(ctx, q, vectorLiteral(vec), f.ProjectKey, topK)
	if err != nil {
		return nil, fmt.Errorf("pgvector: query: %w", err)
	}
	defer rows.Close()

	var hits []Hit
	for rows.Next() {
		var h Hit
		if err := rows.Scan(&h.IssueKey, &h.Summary, &h.Status, &h.Score, &h.Excerpt); err != nil {
			return nil, err
		}
		hits = append(hits, h)
	}
	return hits, rows.Err()
}

func (s *PgvectorStore) Stats(ctx context.Context, projectKey string) (int, error) {
	var n int
	err := s.pool.QueryRow(ctx, `SELECT count(*) FROM issues_index WHERE project_key = $1`, projectKey).Scan(&n)
	return n, err
}
```

```go
package index

import (
	"context"
	"fmt"
	"strings"

	"github.com/grevus/mcp-jira/internal/jira"
	"github.com/grevus/mcp-jira/internal/rag/embed"
	"github.com/grevus/mcp-jira/internal/rag/store"
)

type IssueDocReader interface {
	IterateIssueDocs(ctx context.Context, projectKey string) (<-chan jira.IssueDoc, <-chan error)
}

type Stats struct {
	Indexed int
}

type Indexer struct {
	Reader   IssueDocReader
	Embedder embed.Embedder
	Store    store.Store
}

func New(r IssueDocReader, e embed.Embedder, s store.Store) *Indexer {
	return &Indexer{Reader: r, Embedder: e, Store: s}
}

// Reindex выполняет полную переиндексацию проекта:
// 1) собирает все IssueDoc через Reader;
// 2) рендерит в плоский текст;
// 3) эмбеддит батчами по 64;
// 4) одним вызовом ReplaceProject атомарно заменяет коллекцию.
func (i *Indexer) Reindex(ctx context.Context, projectKey string) (Stats, error) {
	docsCh, errsCh := i.Reader.IterateIssueDocs(ctx, projectKey)

	var collected []jira.IssueDoc
	for d := range docsCh {
		collected = append(collected, d)
	}
	if err, ok := <-errsCh; ok && err != nil {
		return Stats{}, fmt.Errorf("indexer: read: %w", err)
	}

	if len(collected) == 0 {
		if err := i.Store.ReplaceProject(ctx, projectKey, nil); err != nil {
			return Stats{}, err
		}
		return Stats{Indexed: 0}, nil
	}

	const batchSize = 64
	storeDocs := make([]store.Document, 0, len(collected))
	for start := 0; start < len(collected); start += batchSize {
		end := start + batchSize
		if end > len(collected) {
			end = len(collected)
		}
		batch := collected[start:end]
		texts := make([]string, len(batch))
		for j, d := range batch {
			texts[j] = RenderDoc(d)
		}
		vecs, err := i.Embedder.Embed(ctx, texts)
		if err != nil {
			return Stats{}, fmt.Errorf("indexer: embed: %w", err)
		}
		if len(vecs) != len(batch) {
			return Stats{}, fmt.Errorf("indexer: embedder returned %d vectors for %d texts", len(vecs), len(batch))
		}
		for j, d := range batch {
			storeDocs = append(storeDocs, store.Document{
				ProjectKey: d.ProjectKey,
				IssueKey:   d.Key,
				Summary:    d.Summary,
				Status:     d.Status,
				Assignee:   d.Assignee,
				Content:    texts[j],
				Embedding:  vecs[j],
			})
		}
	}

	if err := i.Store.ReplaceProject(ctx, projectKey, storeDocs); err != nil {
		return Stats{}, fmt.Errorf("indexer: replace: %w", err)
	}
	return Stats{Indexed: len(storeDocs)}, nil
}

// RenderDoc — детерминированный рендер IssueDoc в плоский текст для embeddings.
// Формат стабилен и покрыт тестом; менять вместе с тестом.
func RenderDoc(d jira.IssueDoc) string {
	var b strings.Builder
	fmt.Fprintf(&b, "[KEY] %s\n", d.Key)
	fmt.Fprintf(&b, "Summary: %s\n", d.Summary)
	fmt.Fprintf(&b, "Status: %s\n", d.Status)
	if d.Assignee != "" {
		fmt.Fprintf(&b, "Assignee: %s\n", d.Assignee)
	}
	b.WriteString("\nDescription:\n")
	b.WriteString(d.Description)
	b.WriteByte('\n')

	if len(d.Comments) > 0 {
		b.WriteString("\nComments:\n")
		for _, c := range d.Comments {
			fmt.Fprintf(&b, "- %s\n", c)
		}
	}
	if len(d.StatusHistory) > 0 {
		b.WriteString("\nStatus history:\n")
		for _, h := range d.StatusHistory {
			fmt.Fprintf(&b, "- %s\n", h)
		}
	}
	if len(d.LinkedIssues) > 0 {
		fmt.Fprintf(&b, "\nLinked issues: %s\n", strings.Join(d.LinkedIssues, ", "))
	}
	return b.String()
}
```

```go
import (
	"context"
	"fmt"

	"github.com/grevus/mcp-jira/internal/rag/embed"
	"github.com/grevus/mcp-jira/internal/rag/store"
)

type Query struct {
	ProjectKey string
	Text       string
	TopK       int
}

type Retriever struct {
	Embedder embed.Embedder
	Store    store.Store
}

func New(e embed.Embedder, s store.Store) *Retriever {
	return &Retriever{Embedder: e, Store: s}
}

func (r *Retriever) Search(ctx context.Context, q Query) ([]store.Hit, error) {
	if q.ProjectKey == "" {
		return nil, fmt.Errorf("retriever: project_key is required")
	}
	if q.Text == "" {
		return nil, fmt.Errorf("retriever: query text is required")
	}
	topK := q.TopK
	if topK <= 0 {
		topK = 5
	}
	if topK > 20 {
		topK = 20
	}

	vecs, err := r.Embedder.Embed(ctx, []string{q.Text})
	if err != nil {
		return nil, fmt.Errorf("retriever: embed: %w", err)
	}
	if len(vecs) != 1 {
		return nil, fmt.Errorf("retriever: expected 1 vector, got %d", len(vecs))
	}
	hits, err := r.Store.Query(ctx, vecs[0], topK, store.Filter{ProjectKey: q.ProjectKey})
	if err != nil {
		return nil, fmt.Errorf("retriever: query: %w", err)
	}
	return hits, nil
}
```

```go
// На ошибке возвращает *mcp.CallToolResult{IsError:true, TextContent(err)}
// и nil в error-слоте — согласно MCP: "ошибка бизнес-логики != transport error".
func adapt[In, Out any](h handlers.Handler[In, Out]) func(context.Context, *mcp.CallToolRequest, In) (*mcp.CallToolResult, Out, error) {
	return func(ctx context.Context, _ *mcp.CallToolRequest, in In) (*mcp.CallToolResult, Out, error) {
		out, err := h(ctx, in)
		if err != nil {
			var zero Out
			return &mcp.CallToolResult{
				IsError: true,
				Content: []mcp.Content{&mcp.TextContent{Text: err.Error()}},
			}, zero, nil
		}
		b, mErr := json.Marshal(out)
		if mErr != nil {
			var zero Out
			return &mcp.CallToolResult{
				IsError: true,
				Content: []mcp.Content{&mcp.TextContent{Text: "marshal error: " + mErr.Error()}},
			}, zero, nil
		}
		return &mcp.CallToolResult{
			Content: []mcp.Content{&mcp.TextContent{Text: string(b)}},
		}, out, nil
```

```go
// Package main — бинарь mcp-jira.
// ВНИМАНИЕ: в stdio-режиме stdout занят JSON-RPC MCP. НИКОГДА не писать в os.Stdout
// через fmt.Println/Printf. log.* пишет в stderr — это OK.
package main

import (
	"context"
	"flag"
	"log"
	"net/http"
	"os"
	"os/signal"
	"syscall"

	"github.com/grevus/mcp-jira/internal/auth"
	"github.com/grevus/mcp-jira/internal/config"
	"github.com/grevus/mcp-jira/internal/jira"
	"github.com/grevus/mcp-jira/internal/rag/embed"
	"github.com/grevus/mcp-jira/internal/rag/retriever"
	"github.com/grevus/mcp-jira/internal/rag/store"
	"github.com/grevus/mcp-jira/internal/register"

	"github.com/labstack/echo/v4"
	"github.com/labstack/echo/v4/middleware"
	"github.com/modelcontextprotocol/go-sdk/mcp"
)

func main() {
	transport := flag.String("transport", "stdio", "stdio|http")
	flag.Parse()

	var mode config.Mode
	switch *transport {
	case "stdio":
		mode = config.ModeStdio
	case "http":
		mode = config.ModeHTTP
	default:
		log.Fatalf("unknown transport %q (want stdio|http)", *transport)
	}

	cfg, err := config.Load(mode)
	if err != nil {
		log.Fatalf("config: %v", err)
	}

	ctx, cancel := signal.NotifyContext(context.Background(), syscall.SIGINT, syscall.SIGTERM)
	defer cancel()

	jc := jira.NewHTTPClient(cfg.JiraBaseURL, cfg.JiraEmail, cfg.JiraAPIToken)

	var embedder embed.Embedder
	switch cfg.RAGEmbedder {
	case "voyage":
		embedder = embed.NewVoyageEmbedder(cfg.VoyageAPIKey, "voyage-3", 1024)
	case "openai":
		embedder = embed.NewOpenAIEmbedder(cfg.OpenAIAPIKey, "text-embedding-3-small", 1024)
	}

	pg, err := store.NewPgvectorStore(ctx, cfg.DatabaseURL)
	if err != nil {
		log.Fatalf("pgvector: %v", err)
	}
	defer pg.Close()

	ret := retriever.New(embedder, pg)

	srv := mcp.NewServer(&mcp.Implementation{
		Name:    "mcp-jira",
		Version: "0.1.0",
	}, nil)
	register.Register(srv, jc, ret)

	switch mode {
	case config.ModeStdio:
		log.Println("mcp-jira: stdio transport")
		if err := srv.Run(ctx, &mcp.StdioTransport{}); err != nil {
			log.Fatalf("stdio: %v", err)
		}
	case config.ModeHTTP:
		e := echo.New()
		e.HideBanner = true
		e.Use(middleware.Recover())
		e.Use(middleware.Logger())

		e.GET("/health", func(c echo.Context) error {
			return c.JSON(http.StatusOK, map[string]string{"status": "ok"})
		})

		handler := mcp.NewStreamableHTTPHandler(
			func(*http.Request) *mcp.Server { return srv },
			nil,
		)
		protected := e.Group("", echo.WrapMiddleware(auth.Middleware(cfg.MCPAPIKey)))
		protected.Any("/mcp", echo.WrapHandler(handler))

		log.Printf("mcp-jira: streamable-http on %s", cfg.MCPAddr)
		go func() {
			<-ctx.Done()
			_ = e.Shutdown(context.Background())
		}()
		if err := e.Start(cfg.MCPAddr); err != nil && err != http.ErrServerClosed {
			log.Fatalf("http: %v", err)
		}
	}
}
```

```go
// Package main — CLI mcp-jira-index: миграции БД и полная переиндексация проекта.
package main

import (
	"context"
	"database/sql"
	"flag"
	"fmt"
	"log"
	"os"
	"os/signal"
	"syscall"
	"time"

	"github.com/grevus/mcp-jira/internal/config"
	"github.com/grevus/mcp-jira/internal/jira"
	"github.com/grevus/mcp-jira/internal/rag/embed"

	"github.com/grevus/mcp-jira/internal/rag/store"

	_ "github.com/jackc/pgx/v5/stdlib"
)

func main() {
	if len(os.Args) < 2 {
		usage()
	}
	sub := os.Args[1]
	args := os.Args[2:]

	cfg, err := config.Load(config.ModeIndex)
	if err != nil {
		log.Fatalf("config: %v", err)
	}

	ctx, cancel := signal.NotifyContext(context.Background(), syscall.SIGINT, syscall.SIGTERM)
	defer cancel()

	switch sub {
	case "migrate":
		runMigrate(ctx, cfg)
	case "index":
		runIndex(ctx, cfg, args)
	case "-h", "--help", "help":
		usage()
	default:
		fmt.Fprintf(os.Stderr, "unknown subcommand %q\n", sub)
		usage()
	}
}

func usage() {
	fmt.Fprintln(os.Stderr, `usage:
  mcp-jira-index migrate                        # apply goose migrations
  mcp-jira-index index --project=KEY            # full reindex of a project`)
	os.Exit(2)
}

func runMigrate(ctx context.Context, cfg config.Config) {
	db, err := sql.Open("pgx", cfg.DatabaseURL)
	if err != nil {
		log.Fatalf("open db: %v", err)
	}
	defer db.Close()

	if err := store.Migrate(ctx, db); err != nil {
		log.Fatalf("migrate: %v", err)
	}
	log.Println("migrations applied")
}

func runIndex(ctx context.Context, cfg config.Config, args []string) {
	fs := flag.NewFlagSet("index", flag.ExitOnError)
	project := fs.String("project", "", "Jira project key (required)")
	_ = fs.Parse(args)
	if *project == "" {
		log.Fatal("index: --project is required")
	}

	jc := jira.NewHTTPClient(cfg.JiraBaseURL, cfg.JiraEmail, cfg.JiraAPIToken)

	var embedder embed.Embedder
	switch cfg.RAGEmbedder {
	case "voyage":
		embedder = embed.NewVoyageEmbedder(cfg.VoyageAPIKey, "voyage-3", 1024)
	case "openai":
		embedder = embed.NewOpenAIEmbedder(cfg.OpenAIAPIKey, "text-embedding-3-small", 1024)
	}

	pg, err := store.NewPgvectorStore(ctx, cfg.DatabaseURL)
	if err != nil {
		log.Fatalf("pgvector: %v", err)
	}
	defer pg.Close()

	idx := index.New(jc, embedder, pg)
	start := time.Now()
	stats, err := idx.Reindex(ctx, *project)
	if err != nil {
		log.Fatalf("reindex: %v", err)
	}
	log.Printf("indexed %d docs for project=%s in %s using embedder=%s",
		stats.Indexed, *project, time.Since(start), embedder.Name())
}
```
