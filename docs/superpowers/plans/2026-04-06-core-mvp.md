# MCP Jira Core MVP — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Поднять рабочий MCP-сервер на Go для Jira с набором базовых tools, API-key аутентификацией, SSE и stdio транспортами — то, что разработчик может локально подключить к Claude Desktop/Cursor и сам начать пользоваться.

**Architecture:** Single-binary Go-сервис на официальном `github.com/modelcontextprotocol/go-sdk`. Слои: `cmd/server` (entrypoint, Echo HTTP-сервер, выбор транспорта), `internal/jira` (HTTP-клиент к Jira REST API v3), `internal/handlers` (чистые функции-хендлеры MCP tools, без знания о сервере и фреймворке), `internal/register` (склейка: принимает `*mcp.Server` и `jira.Client`, регистрирует все хендлеры под именами и описаниями MCP tools), `internal/auth` (stdlib middleware валидации API-ключа), `internal/config` (env-конфигурация). Тесты — табличные `*_test.go` рядом с кодом, Jira замокан через интерфейс.

**Tech Stack:** Go 1.23, `github.com/modelcontextprotocol/go-sdk`, `github.com/labstack/echo/v4` (HTTP-роутинг и middleware для SSE/health), `net/http`, стандартный `testing` + `github.com/stretchr/testify/require`.

**Out of scope (отдельные планы):** Postgres-хранилище ключей, Stripe metered billing, Dockerfile/деплой на Railway, листинги, observability/Prometheus, landing page. Здесь — только локально запускаемый и тестируемый сервер.

package jira

import (
	"context"
	"encoding/json"
	"fmt"
	"net/http"
	"net/url"
	"strings"
	"time"
)

type ListIssuesParams struct {
	ProjectKey string
	Status     string
	Assignee   string
}

type Client interface {
	ListIssues(ctx context.Context, p ListIssuesParams) ([]Issue, error)
	GetSprintHealth(ctx context.Context, boardID int) (SprintHealth, error)
}

type HTTPClient struct {
	baseURL string
	email   string
	token   string
	http    *http.Client
}

func NewHTTPClient(baseURL, email, token string) *HTTPClient {
	return &HTTPClient{
		baseURL: strings.TrimRight(baseURL, "/"),
		email:   email,
		token:   token,
		http:    &http.Client{Timeout: 15 * time.Second},
	}
}

func (c *HTTPClient) do(ctx context.Context, method, path string, q url.Values, out any) error {
	u := c.baseURL + path
	if len(q) > 0 {
		u += "?" + q.Encode()
	}
	req, err := http.NewRequestWithContext(ctx, method, u, nil)
	if err != nil {
		return err
	}
	req.SetBasicAuth(c.email, c.token)
	req.Header.Set("Accept", "application/json")
	resp, err := c.http.Do(req)
	if err != nil {
		return err
	}
	defer resp.Body.Close()
	if resp.StatusCode >= 400 {
		return fmt.Errorf("jira: %s %s -> %d", method, path, resp.StatusCode)
	}
	return json.NewDecoder(resp.Body).Decode(out)
}

func (c *HTTPClient) ListIssues(ctx context.Context, p ListIssuesParams) ([]Issue, error) {
	parts := []string{fmt.Sprintf(`project = %q`, p.ProjectKey)}
	if p.Status != "" {
		parts = append(parts, fmt.Sprintf(`status = %q`, p.Status))
	}
	if p.Assignee != "" {
		parts = append(parts, fmt.Sprintf(`assignee = %q`, p.Assignee))
	}
	q := url.Values{"jql": {strings.Join(parts, " AND ")}}

	var raw struct {
		Issues []struct {
			Key    string `json:"key"`
			Fields struct {
				Summary string `json:"summary"`
				Status  struct {
					Name string `json:"name"`
				} `json:"status"`
				Assignee *struct {
					DisplayName string `json:"displayName"`
				} `json:"assignee"`
			} `json:"fields"`
		} `json:"issues"`
	}
	if err := c.do(ctx, http.MethodGet, "/rest/api/3/search", q, &raw); err != nil {
		return nil, err
	}
	out := make([]Issue, 0, len(raw.Issues))
	for _, it := range raw.Issues {
		issue := Issue{Key: it.Key, Summary: it.Fields.Summary, Status: it.Fields.Status.Name}
		if it.Fields.Assignee != nil {
			issue.Assignee = it.Fields.Assignee.DisplayName
		}
		out = append(out, issue)
	}
	return out, nil
}

func (c *HTTPClient) GetSprintHealth(ctx context.Context, boardID int) (SprintHealth, error) {

package auth

import (
	"crypto/subtle"
	"net/http"
)

func Middleware(expected string) func(http.Handler) http.Handler {
	exp := []byte(expected)
	return func(next http.Handler) http.Handler {
		return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
			got := []byte(r.Header.Get("X-API-Key"))
			if len(got) == 0 || subtle.ConstantTimeCompare(got, exp) != 1 {
				http.Error(w, "unauthorized", http.StatusUnauthorized)
				return
			}
			next.ServeHTTP(w, r)
		})
	}
}
```

func (c *HTTPClient) GetSprintHealth(ctx context.Context, boardID int) (SprintHealth, error) {
	var sprintsResp struct {
		Values []struct {
			ID   int    `json:"id"`
			Name string `json:"name"`
		} `json:"values"`
	}
	q := url.Values{"state": {"active"}}
	path := fmt.Sprintf("/rest/agile/1.0/board/%d/sprint", boardID)
	if err := c.do(ctx, http.MethodGet, path, q, &sprintsResp); err != nil {
		return SprintHealth{}, err
	}
	if len(sprintsResp.Values) == 0 {
		return SprintHealth{}, fmt.Errorf("jira: no active sprint for board %d", boardID)
	}
	sp := sprintsResp.Values[0]

	var issuesResp struct {
		Issues []struct {
			Fields struct {
				Status struct {
					Name string `json:"name"`
				} `json:"status"`
			} `json:"fields"`
		} `json:"issues"`
	}
	if err := c.do(ctx, http.MethodGet, fmt.Sprintf("/rest/agile/1.0/sprint/%d/issue", sp.ID), nil, &issuesResp); err != nil {
		return SprintHealth{}, err
	}

	h := SprintHealth{BoardID: boardID, SprintName: sp.Name, Total: len(issuesResp.Issues)}
	for _, it := range issuesResp.Issues {
		switch strings.ToLower(it.Fields.Status.Name) {
		case "done", "closed", "resolved":
			h.Done++
		case "in progress":
			h.InProgress++
		case "blocked":
			h.Blocked++
		}
	}
	if h.Total > 0 {
		h.Velocity = float64(h.Done) / float64(h.Total)
	}
	return h, nil
}
```

```go
package main

import (
	"flag"
	"log"
	"net/http"

	"github.com/grevus/mcp-jira/internal/auth"
	"github.com/grevus/mcp-jira/internal/config"
	"github.com/grevus/mcp-jira/internal/jira"
	"github.com/grevus/mcp-jira/internal/register"
	"github.com/labstack/echo/v4"
	"github.com/labstack/echo/v4/middleware"
	"github.com/modelcontextprotocol/go-sdk/mcp"
)

func main() {
	transport := flag.String("transport", "stdio", "stdio|sse")
	flag.Parse()

	cfg, err := config.Load()
	if err != nil {
		log.Fatalf("config: %v", err)
	}

	jc := jira.NewHTTPClient(cfg.JiraBaseURL, cfg.JiraEmail, cfg.JiraAPIToken)

	srv := mcp.NewServer(
		mcp.WithName("mcp-jira"),
		mcp.WithVersion("0.1.0"),
	)
	register.Register(srv, jc)

	switch *transport {
	case "stdio":
		log.Println("mcp-jira: stdio transport")
		if err := srv.ServeStdio(); err != nil {
			log.Fatal(err)
		}
	case "sse":
		e := echo.New()
		e.HideBanner = true
		e.Use(middleware.Recover())
		e.Use(middleware.Logger())

		// Health — без auth.
		e.GET("/health", func(c echo.Context) error {
			return c.JSON(http.StatusOK, map[string]string{"status": "ok"})
		})

		// SSE-группа с API-key middleware.
		// auth.Middleware — стандартный func(http.Handler) http.Handler,
		// поэтому оборачиваем через echo.WrapMiddleware.
		// SSEHandler от MCP SDK — стандартный http.Handler, монтируем через echo.WrapHandler.
		sseGroup := e.Group("", echo.WrapMiddleware(auth.Middleware(cfg.MCPAPIKey)))
		sseGroup.GET("/sse", echo.WrapHandler(srv.SSEHandler()))
		sseGroup.POST("/sse", echo.WrapHandler(srv.SSEHandler()))

		log.Printf("mcp-jira: sse transport on %s", cfg.MCPAddr)
		if err := e.Start(cfg.MCPAddr); err != nil && err != http.ErrServerClosed {
			log.Fatal(err)
		}
	default:
		log.Fatalf("unknown transport %q", *transport)
	}
}
```

> Если имена методов SDK (`ServeStdio`, `SSEHandler`, `WithName`, `AddTool`) отличаются от текущей версии — это единственное место плюс `internal/register/register.go`, где их надо подправить. Поведение хендлеров и тесты на них не меняются.
>
> `echo.WrapHandler` приводит `http.Handler` к `echo.HandlerFunc`, `echo.WrapMiddleware` — `func(http.Handler) http.Handler` к `echo.MiddlewareFunc`. Это позволяет переиспользовать stdlib-совместимый `auth.Middleware` без переписывания на echo-специфичную сигнатуру.
